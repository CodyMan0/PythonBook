### 입력

```py
## 가운데를 기점으로 양 문자열의 합이 같으면 LUCKY 출력 nor READY

# 내풀이
n = int(input())
condition = input()

print(n//2)
left = sum(list(map(int,condition[:n//2])))
right = sum(list(map(int,condition[n//2:])))

if left == right :
print("LUCKY")
else :
print("READY")

    # 책풀이

n = input()
length = len(n)
summary = 0

for i in range(length // 2) :
summary += int(n[i])

for i in range(length // 2, length) :
summary -= int(n[i])

if summary == 0 :
print("LUCKY")
else :
print("READY")
```
